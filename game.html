let clock = new THREE.Clock(); let ready = { player: false, car: false }; function animate(){ requestAnimationFrame(animate); let dt = Math.min(clock.getDelta(), 0.033); if(!player || !car) { renderer.render(scene,camera); return; } // ===================== // WALK MODE // ===================== if(!inCar){ let speed = 4 * dt; if(keys["w"]) player.position.z -= speed; if(keys["s"]) player.position.z += speed; if(keys["a"]) player.position.x -= speed; if(keys["d"]) player.position.x += speed; } // ===================== // CAR MODE (FIXED PHYSICS FEEL) // ===================== if(inCar){ let accel = 6 * dt; let turn = 2.5 * dt; if(keys["w"]) car.userData.speed = (car.userData.speed || 0) + accel; if(keys["s"]) car.userData.speed = (car.userData.speed || 0) - accel; if(keys["a"]) car.rotation.y += turn; if(keys["d"]) car.rotation.y -= turn; // friction car.userData.speed *= Math.pow(0.92, dt * 60); car.position.x += Math.sin(car.rotation.y) * car.userData.speed; car.position.z += Math.cos(car.rotation.y) * car.userData.speed; // player sticks to car (clean seat system) player.position.set( car.position.x, car.position.y + 1, car.position.z ); } // ===================== // ENTER / EXIT FIXED // ===================== if(keys["e"] && !eLast){ if(!inCar && car && player.position.distanceTo(car.position) < 3){ inCar = true; } else if(inCar){ inCar = false; player.position.x += Math.sin(car.rotation.y) * 2; player.position.z += Math.cos(car.rotation.y) * 2; } } eLast = keys["e"]; // ===================== // CAMERA (FIXED GTA STYLE FOLLOW) // ===================== let target = inCar ? car.position : player.position; let desired = new THREE.Vector3( target.x, 5, target.z + 8 ); cam.lerp(desired, 0.1); camera.position.copy(cam); camera.lookAt(target); renderer.render(scene,camera); } animate();
