let clock = new THREE.Clock(); let carSpeed = 0; let cam = new THREE.Vector3(); function animate(){ requestAnimationFrame(animate); let dt = Math.min(clock.getDelta(), 0.033); if(!player || !car){ renderer.render(scene,camera); return; } // ===================== // WALK MODE // ===================== if(!inCar){ let speed = 4 * dt; if(keys["w"]) player.position.z -= speed; if(keys["s"]) player.position.z += speed; if(keys["a"]) player.position.x -= speed; if(keys["d"]) player.position.x += speed; } // ===================== // CAR MODE (IMPROVED GTA FEEL) // ===================== if(inCar){ let accel = 6 * dt; let turn = 2.8 * dt; // acceleration / braking if(keys["w"]) carSpeed += accel; if(keys["s"]) carSpeed -= accel; // natural friction (VERY IMPORTANT FOR GTA FEEL) if(!keys["w"] && !keys["s"]){ carSpeed *= Math.pow(0.90, dt * 60); } // clamp speed (prevents breaking physics) carSpeed = Math.max(-6, Math.min(12, carSpeed)); // steering depends on speed (GTA-style feel) let steerFactor = Math.min(1, Math.abs(carSpeed) / 5); if(keys["a"]) car.rotation.y += turn * steerFactor; if(keys["d"]) car.rotation.y -= turn * steerFactor; // movement vector (FIXED physics style) let forward = new THREE.Vector3( Math.sin(car.rotation.y), 0, Math.cos(car.rotation.y) ); car.position.add(forward.multiplyScalar(carSpeed * dt)); // smoother player follow (NOT snapping) let seatOffset = forward.clone().multiplyScalar(-0.8); player.position.lerp( car.position.clone().add(seatOffset).setY(1), 0.2 ); } // ===================== // ENTER / EXIT (FIXED) // ===================== if(keys["e"] && !eLast){ if(!inCar && car && player.position.distanceTo(car.position) < 3){ inCar = true; } else if(inCar){ inCar = false; let exitOffset = new THREE.Vector3( Math.sin(car.rotation.y) * 2, 0, Math.cos(car.rotation.y) * 2 ); player.position.add(exitOffset); } } eLast = keys["e"]; // ===================== // CAMERA (GTA STYLE FOLLOW FIX) // ===================== let target = inCar ? car.position : player.position; let offset = new THREE.Vector3( target.x, 5, target.z + 8 ); cam.lerp(offset, 0.1); camera.position.copy(cam); camera.lookAt(target); renderer.render(scene,camera); } animate();
